# TITLE
Cursor

# SHORT
Select the mouse cursor to show when the mouse pointer is over the element.

# LONG
Select the mouse cursor to show when the mouse pointer is over the element.

# SYNTAX

cursor: auto;

# SYNTAX

# SYNTAX
